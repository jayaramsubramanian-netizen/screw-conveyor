# ── Materials CRUD ─────────────────────────────────────────────────────────
@app.get("/db/materials")
def list_materials(search: str = ""):
    if search:
        rows = db_query("SELECT * FROM materials WHERE name LIKE ? ORDER BY name",(f"%{search}%",))
    else:
        rows = db_query("SELECT * FROM materials ORDER BY name")
    return {"count":len(rows),"materials":rows}

@app.get("/db/materials/{name}")
def get_material_route(name: str):
    rows = db_query("SELECT * FROM materials WHERE name=?",(name,))
    if not rows: raise HTTPException(404,f"'{name}' not found")
    return rows[0]

class MaterialCreate(BaseModel):
    name: str; bulk_density: float; lambda_cema: float=1.5; lambda_kws: float=1.55
    lambda_din: float=1.5; trough_loading: float=0.30; Ks: float=1.0; abrasive: str="Low"
    wear_coeff: float=1.0; cema_class: str="II"; angle_of_repose: float=35.0
    particle_size_mm: float=10.0; moisture_pct: float=0.0; temperature_c: float=20.0; notes: str=""

@app.post("/db/materials", status_code=201)
def create_material(mat: MaterialCreate):
    if db_query("SELECT id FROM materials WHERE name=?",(mat.name,)):
        raise HTTPException(409,f"'{mat.name}' already exists")
    lid = db_exec("""INSERT INTO materials (name,bulk_density,lambda_cema,lambda_kws,lambda_din,
        trough_loading,Ks,abrasive,wear_coeff,cema_class,angle_of_repose,particle_size_mm,
        moisture_pct,temperature_c,notes,custom) VALUES (?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,1)""",
        (mat.name,mat.bulk_density,mat.lambda_cema,mat.lambda_kws,mat.lambda_din,
         mat.trough_loading,mat.Ks,mat.abrasive,mat.wear_coeff,mat.cema_class,
         mat.angle_of_repose,mat.particle_size_mm,mat.moisture_pct,mat.temperature_c,mat.notes))
    return {"id":lid,"message":f"Material '{mat.name}' created"}

@app.put("/db/materials/{name}")
def update_material(name: str, mat: MaterialCreate):
    if not db_query("SELECT id FROM materials WHERE name=?",(name,)): raise HTTPException(404)
    db_exec("""UPDATE materials SET bulk_density=?,lambda_cema=?,lambda_kws=?,lambda_din=?,
        trough_loading=?,Ks=?,abrasive=?,wear_coeff=?,cema_class=?,angle_of_repose=?,
        particle_size_mm=?,moisture_pct=?,temperature_c=?,notes=?,custom=1,updated_at=datetime('now')
        WHERE name=?""",
        (mat.bulk_density,mat.lambda_cema,mat.lambda_kws,mat.lambda_din,mat.trough_loading,
         mat.Ks,mat.abrasive,mat.wear_coeff,mat.cema_class,mat.angle_of_repose,
         mat.particle_size_mm,mat.moisture_pct,mat.temperature_c,mat.notes,name))
    return {"message":f"'{name}' updated"}

@app.delete("/db/materials/{name}")
def delete_material(name: str):
    rows = db_query("SELECT custom FROM materials WHERE name=?",(name,))
    if not rows: raise HTTPException(404)
    if not rows[0]["custom"]: raise HTTPException(403,"Cannot delete built-in material")
    db_exec("DELETE FROM materials WHERE name=?",(name,))
    return {"message":"Deleted"}

# ── Conveyor Calculations ──────────────────────────────────────────────────
@app.post("/calculate")
def calculate(inp:ConveyorInput):
    try:
        eng = ENGINE_MAP.get(inp.standard)
        if not eng: raise HTTPException(400,f"Unknown: {inp.standard}")
        return eng.calculate(inp)
    except ValueError as e: raise HTTPException(400,str(e))
    except Exception as e: raise HTTPException(500,traceback.format_exc())

@app.post("/calculate/all")
def calculate_all(inp:ConveyorInput):
    results={}
    for std,eng in ENGINE_MAP.items():
        try: results[std]=eng.calculate(inp.model_copy(update={"standard":std})).model_dump()
        except Exception as e: results[std]={"error":str(e)}
    return results

@app.post("/sweep/speed")
def sweep_speed(inp:ConveyorInput):
    eng=ENGINE_MAP.get(inp.standard,cema_engine); out=[]
    for N in range(10,210,10):
        try:
            r=eng.calculate(inp.model_copy(update={"speed_rpm":float(N)}))
            out.append({"speed_rpm":N,"capacity_t_h":r.capacity.achieved_t_h,
                        "power_kW":r.power.P_total_kW,"motor_kW":r.power.motor_selected_kW,
                        "torque_Nm":r.torque.running_Nm})
        except: pass
    return {"sweep":out,"standard":inp.standard}

@app.post("/sweep/diameter")
def sweep_diameter(inp:ConveyorInput):
    eng=ENGINE_MAP.get(inp.standard,cema_engine); out=[]
    for D in [0.10,0.15,0.20,0.25,0.30,0.35,0.40,0.45,0.50,0.60]:
        try:
            r=eng.calculate(inp.model_copy(update={"trough_diameter_m":D,"pitch_m":D}))
            out.append({"diameter_mm":int(D*1000),"capacity_t_h":r.capacity.achieved_t_h,
                        "power_kW":r.power.P_total_kW,"cost_usd":r.cost.estimated_cost_usd})
        except: pass
    return {"sweep":out,"standard":inp.standard}

@app.post("/sweep/length")

@app.post("/family/design")
def family_design_route(inp:FamilyDesignInput):
    try: return design_family(inp)
    except Exception as e: raise HTTPException(500,str(e))

@app.post("/family/matrix")
def family_matrix(inp:FamilyDesignInput):
    family=design_family(inp); matrix={}
    for pt in family.points:
        key=f"{pt.diameter_mm}mm"
        if key not in matrix: matrix[key]={}
        matrix[key][f"{pt.length_m}m"]={"capacity":pt.max_capacity_t_h,"power":pt.power_kW,
            "motor":pt.motor_kW,"cost":pt.cost_usd,"L10":pt.L10_hours}
    return {"matrix":matrix,"diameters":[f"{d}mm" for d in inp.diameters_mm],
            "lengths":[f"{l}m" for l in inp.lengths_m],"material":inp.material,"standard":inp.standard}
